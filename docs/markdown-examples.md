export const AlertBox = ({children, type}) => {
  const colors = {
    success: '#4caf50',
    error: '#f44336',
    warning: '#ff9800',

      padding: '15px',
      borderRadius: '5px',
      marginTop: '15px',
      marginBottom: '15px',
    }}>
      {children}
    </div>
  );
};
